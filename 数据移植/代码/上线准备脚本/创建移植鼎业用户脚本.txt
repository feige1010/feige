  identified by DINGYEV1_BANK

  temporary tablespace TEMP
  profile DEFAULT;

grant dba to DINGYEV1_BANK;
-- Grant/Revoke system privileges 
grant select any table to DINGYEV1_BANK;
grant unlimited tablespace to DINGYEV1_BANK;
